Basics & Variables

print("Hello, World!")
name = "JometCode"
age = 2
pi = 3.14
is_active = True
nothing = None
# Dynamic typing — type changes freely
x = 5    # int
x = "hi"  # now str
# Multiple assignment
a, b, c = 1, 2, 3
# Type checking
print(type(age))  # <class 'int'>

Strings

s = "Python"
s2 = 'Python'  # single quotes work too
msg = f"Hello {s}"           # f-string (Python 3.6+)
msg2 = f"Sum: {2 + 3}"        # expressions inside
upper = s.upper()              # "PYTHON"
lower = s.lower()              # "python"
stripped = "  hi  ".strip()     # "hi"
words = "a,b,c".split(",")    # ["a", "b", "c"]
joined = ", ".join(words)     # "a, b, c"
replaced = s.replace("Py", "Ji") # "Jithon"
length = len(s)                 # 6
is_in = "th" in s               # True

Lists & Tuples

# Lists (mutable)
nums = [1, 2, 3, 4, 5]
nums.append(6)       # [1,2,3,4,5,6]
nums.insert(0, 0)    # [0,1,2,3,4,5,6]
nums.pop()            # removes & returns last
nums.sort()           # in-place sort
nums.reverse()        # in-place reverse
sliced = nums[1:3]     # [2,3] (end exclusive)
sliced = nums[:3]      # [1,2,3]
sliced = nums[::2]     # [1,3,5] (step)
# Tuples (immutable)
point = (3, 4)
x, y = point          # unpacking

Dictionaries & Sets

# Dict (key-value pairs)
user = {"name": "Alice", "age": 30}
user["email"] = "a@b.com"   # add key
name = user.get("name", "default")
for k, v in user.items(): print(k, v)
keys = user.keys()          # view of keys
vals = user.values()        # view of values
# Set (unique, unordered)
a = {1, 2, 3}
b = {3, 4, 5}
a | b  # union: {1,2,3,4,5}
a & b  # intersection: {3}
a - b  # difference: {1,2}

Control Flow

# If / Elif / Else
if age >= 18:
    print("Adult")
elif age >= 13:
    print("Teen")
else:
    print("Child")

# Ternary
status = "Adult" if age >= 18 else "Minor"

# For loop
for i in range(5):        # 0 to 4
    print(i)

for i, item in enumerate(items):  # with index
    print(i, item)

# While loop
while count < 10:
    count += 1

# Break / Continue
for x in nums:
    if x == 0: continue  # skip
    if x < 0: break     # exit loop

Functions

def greet(name, greeting="Hello"):
    """Return a greeting."""
    return f"{greeting}, {name}!"

# args / kwargs
def log(*args, **kwargs):
    print(args, kwargs)

# Lambda (anonymous)
square = lambda x: x**2
sorted(items, key=lambda i: i["name"])

# Type hints (Python 3.5+)
def add(a: int, b: int) -> int:
    return a + b

Classes & OOP

class Dog:
    species = "Canis familiaris"  # class variable

    def __init__(self, name):
        self.name = name            # instance variable

    def bark(self):
        return f"{self.name} says woof!"

    @classmethod
    def from_birth(cls, year): ...

    @staticmethod
    def info():
        return "Dogs are mammals."

    @property
    def name_upper(self):
        return self.name.upper()

# Inheritance
class Puppy(Dog):
    def bark(self):
        return "yip!"              # override

Error Handling

try:
    n = int(input("Number: "))
    print(10 / n)
except ValueError:
    print("Not a number")
except ZeroDivisionError as e:
    print(e)
else:
    print("Success!")        # runs if no error
finally:
    print("Always runs")    # cleanup

# Raise custom
raise ValueError("Invalid input")

Comprehensions

# List comprehension
squares = [x**2 for x in range(10)]
evens = [x for x in nums if x % 2 == 0]

# Dict comprehension
squares_dict = {x: x**2 for x in range(5)}

# Set comprehension
unique = {len(w) for w in ["hi", "hello", "ok"]}

# Nested comprehension (flatten)
flat = [item for row in matrix for item in row]

File I/O

# Writing
with open("file.txt", "w", encoding="utf-8") as f:
    f.write("Hello!")

# Reading
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()          # entire file
    for line in f:              # line by line
        print(line)

# Modes: "r" read, "w" write, "a" append, "x" exclusive
# Add "b" for binary: "rb", "wb"

# Using pathlib (modern)
from pathlib import Path
Path("file.txt").read_text(encoding="utf-8")

JSON

import json

# Python to JSON string
data = {"name": "Alice", "scores": [90, 85, 88]}
json_str = json.dumps(data, indent=2)

# JSON string to Python
parsed = json.loads(json_str)

# File I/O
with open("data.json", "w") as f: json.dump(data, f)
with open("data.json") as f: data = json.load(f)

Useful Built-in Functions

print(len(items))        # length
print(type(value))        # type of value

print(sorted([3, 1, 2]))  # [1,2,3]
print(enumerate(["a","b"])) # (idx,val) pairs
print(zip([1,2], ["a","b"])) # pairs both lists
